#include <iostream>
#include <string>
#include <memory>
#include "save.h"

using namespace std;

// [문제] 자원을 할당하는 클래스 String (표준 string과 유사한)를 작성하자.
// special 함수의 동작을 관찰하자

class String
{
	size_t len{};
	char* p{ nullptr };

public:
	String( const char* str)
		: len{strlen(str)}
		//, p = new char[len] 순서 보장 X -> 망할수있다.
	{
		p = new char[len];
		memcpy(p, str, len);	//고속 메모리 복사 - DMA
	}

	//지역을 벗어날 때
	//객체가 사라지기 전에 부르는 함수
	~String()
	{
		// if(p != nullptr) 널ptr을 지우는 것은 아무 문제가 없다.
		delete[] p;
	}

	//인터페이스 함수
	size_t size() const 
	{
		return len;
	}

	//프렌드 함수
	friend ostream& operator<<(ostream& os, const String& s)
	{
		for (int i = 0; i < s.len; ++i)
		{
			os <<  s.p[i];
		}
		return os;
	}
};

//-------
int main()
//-------
{
	String s{ "1234567890" };
	cout << s.size() << " - " << s << endl;


	save("소스.cpp");
}
