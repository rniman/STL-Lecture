#include "save.h"

extern bool 관찰;

//-------
{	
	//관찰 = true;

	vector<String> v{ "1", "12", "123" };
	
	관찰 = true;
	// v를 길이기준 내림차순으로 정렬하라
	sort(v.begin(), v.end(), [](const String& a, const String& b) {

		});

	// 출력하라
	for (const String& s : v)
	{
		cout << s << ' ';
	}
