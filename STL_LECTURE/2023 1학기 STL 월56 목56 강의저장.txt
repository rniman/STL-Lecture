#include <array>
#include <fstream>

#include "save.h"
#include "String.h"

using namespace std;

extern bool 관찰;

//-------
int main()
//-------
{	
	//관찰 = true;

	array<String, 30> a;

	ifstream in{ "소스.cpp" };

	if (!in)
	{
		cout << "소스.cpp를 열지 못했습니다." << endl;
		return 0;
	}

	for (int i = 0; i < 30; ++i)
	{
		in >> a[i];
	}

	// a의 길이 기준 오름차순으로 a를 정렬하라
	sort(a.begin(), a.end(), [](const String& a, const String& b) {
		return a.size() < b.size();
		});

	// a의 내용을 모두 출력
	for (const String& s : a)
	{
		cout << s << endl;
	}

	save("소스.cpp");
}

#include <iostream>
#include <array>
#include <fstream>
#include <algorithm>
#include "save.h"
#include "String.h"

using namespace std;

extern bool 관찰;

//-------
int main()
//-------
{	
	// [문제] 키보드에서 입력한 int를 10개만 array를 사용하여 저장하라.
	// 합계를 화면에 출력하라.
	array<int, 10> a;

	for (int& i : a)
	{
		cin >> i;
	}
	cin.clear();
	
	int sum = *a.begin();
	cout << *a.begin();
	for (auto i = a.begin() + 1; i!= a.end(); ++i)
	{
		cout << " + " << *i;
		sum += *i;
	}
	cout << " = " << sum << endl;

	save("소스.cpp");
}
